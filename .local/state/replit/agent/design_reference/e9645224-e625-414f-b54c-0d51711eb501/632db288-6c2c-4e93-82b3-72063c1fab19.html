Database Schema Design

Core Tables

users

id UUID PRIMARY KEY
email VARCHAR UNIQUE
phone VARCHAR UNIQUE
password_hash VARCHAR
user_type ENUM(consumer, business, admin)
is_verified BOOLEAN DEFAULT FALSE
created_at TIMESTAMP
updated_at TIMESTAMP

businesses

id UUID PRIMARY KEY
user_id UUID FOREIGN KEY
business_name VARCHAR
business_type ENUM(restaurant, hotel, bakery, supermarket)
address TEXT
latitude DECIMAL(10,8)
longitude DECIMAL(11,8)
verification_status ENUM(pending, verified, rejected)
rating DECIMAL(3,2)

food_listings

id UUID PRIMARY KEY
business_id UUID FOREIGN KEY
title VARCHAR
description TEXT
original_price DECIMAL(10,2)
discounted_price DECIMAL(10,2)
quantity_available INTEGER
expiry_time TIMESTAMP
pickup_start TIME
pickup_end TIME
status ENUM(active, sold_out, expired)

Transactional Tables

orders

id UUID PRIMARY KEY
user_id UUID FOREIGN KEY
business_id UUID FOREIGN KEY
total_amount DECIMAL(10,2)
status ENUM(pending, confirmed, ready, completed, cancelled)
pickup_time TIMESTAMP
qr_code VARCHAR
created_at TIMESTAMP

order_items

id UUID PRIMARY KEY
order_id UUID FOREIGN KEY
food_listing_id UUID FOREIGN KEY
quantity INTEGER
unit_price DECIMAL(10,2)
subtotal DECIMAL(10,2)

payments

id UUID PRIMARY KEY
order_id UUID FOREIGN KEY
payment_reference VARCHAR UNIQUE
payment_method ENUM(card, transfer, wallet)
amount DECIMAL(10,2)
status ENUM(pending, success, failed)
paystack_reference VARCHAR
processed_at TIMESTAMP

analytics_events

id UUID PRIMARY KEY
event_type VARCHAR
entity_id UUID
entity_type VARCHAR
metadata JSONB
created_at TIMESTAMP

Authentication & Users

User authentication

Get user profile information

PUT /api/users/profile PUT

Update user profile

Business Management

POST /api/businesses POST

Register business profile

GET /api/businesses/:id GET

Get business details

POST /api/businesses/:id/verify POST

Submit verification documents

GET /api/businesses/:id/analytics GET

Business performance analytics

Food Listings

Create new food listing

Browse available food listings
